## Code Pattern Used

```typescript
import { useI18n } from "@/lib/i18n"

export default function TracePage() {
  const { t } = useI18n()
  
  // Basic translation
  return <h1>{t("trace.title")}</h1>
  
  // Translation with parameters
  return <p>{t("trace.progress.complete", { percent: 75 })}</p>
  
  // Conditional translations
  const status = "delivered"
  return <Badge>{t(`trace.status.${status}`)}</Badge>
}
```
